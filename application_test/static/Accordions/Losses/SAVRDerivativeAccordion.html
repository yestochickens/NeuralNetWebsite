\[\nabla_{\mathbf{\hat{Y}}} J \left( \mathbf{\hat{Y}}, \mathbf{Y} \right) = - \mathrm{sgn} \left( \mathbf{Y} - \mathbf{\hat{Y}} \right)\]

\[\frac{\partial}{\partial \mathbf{\hat{Y}}} J \left( \begin{bmatrix} \widehat{y_0} \\ \vdots \\ \widehat{y_{n_y}} \end{bmatrix}, \begin{bmatrix} y_0 \\ \vdots \\ y_{n_y} \end{bmatrix} \right) = - \mathrm{sgn} \left( \begin{bmatrix} y_0 \\ \vdots \\ y_{n_y} \end{bmatrix} - \begin{bmatrix} \widehat{y_0} \\ \vdots \\ \widehat{y_{n_y}} \end{bmatrix} \right)\]

\[\frac{\partial}{\partial \mathbf{\hat{Y}}} J \left( \begin{bmatrix} \widehat{y_0} \\ \vdots \\ \widehat{y_{n_y}} \end{bmatrix}, \begin{bmatrix} y_0 \\ \vdots \\ y_{n_y} \end{bmatrix} \right) = \begin{bmatrix} -\mathrm{sgn} \left( y_0 - \widehat{y_0} \right) \\ \vdots \\ -\mathrm{sgn} \left( y_{n_y} - \widehat{y_{n_y}} \right) \end{bmatrix}\]
