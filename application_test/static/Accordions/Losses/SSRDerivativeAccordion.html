\[\nabla_{\mathbf{\hat{Y}}} J \left( \mathbf{\hat{Y}}, \mathbf{Y} \right) = -2 \left( \mathbf{Y} - \mathbf{\widehat{Y}} \right)\]

\[\frac{\partial}{\partial \mathbf{\hat{Y}}} J \left( \begin{bmatrix} \widehat{y_0} \\ \vdots \\ \widehat{y_{n_y}} \end{bmatrix}, \begin{bmatrix} y_0 \\ \vdots \\ y_{n_y} \end{bmatrix} \right) = -2 \left( \begin{bmatrix} y_0 \\ \vdots \\ y_{n_y} \end{bmatrix} - \begin{bmatrix} \widehat{y_0} \\ \vdots \\ \widehat{y_{n_y}} \end{bmatrix} \right)\]

\[\frac{\partial}{\partial \mathbf{\hat{Y}}} J \left( \begin{bmatrix} \widehat{y_0} \\ \vdots \\ \widehat{y_{n_y}} \end{bmatrix}, \begin{bmatrix} y_0 \\ \vdots \\ y_{n_y} \end{bmatrix} \right) = \begin{bmatrix} 2 \widehat{y_0} - 2 y_0 \\ \vdots \\ 2 \widehat{y_{n_y}} - 2 y_{n_y} \end{bmatrix}\]
